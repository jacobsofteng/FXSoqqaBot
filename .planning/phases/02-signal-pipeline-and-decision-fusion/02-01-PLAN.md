       ```python
       from __future__ import annotations
       import numpy as np
       from dataclasses import dataclass, field
       from datetime import UTC, datetime
       from enum import Enum
       from typing import Protocol, runtime_checkable
       from fxsoqqabot.core.events import DOMSnapshot

       class RegimeState(str, Enum):
           TRENDING_UP = "trending_up"
           TRENDING_DOWN = "trending_down"
           RANGING = "ranging"
           HIGH_CHAOS = "high_chaos"
           PRE_BIFURCATION = "pre_bifurcation"

       @dataclass(frozen=True, slots=True)
       class SignalOutput:
           module_name: str           # "chaos", "flow", "timing"
           direction: float           # -1.0 (sell) to +1.0 (buy), 0.0 = neutral
           confidence: float          # 0.0 to 1.0
           regime: RegimeState | None = None  # Current regime (chaos module only)
           metadata: dict = field(default_factory=dict)
           timestamp: datetime = field(default_factory=lambda: datetime.now(UTC))

       @runtime_checkable
       class SignalModule(Protocol):
           @property
           def name(self) -> str: ...

           async def update(
               self,
               tick_arrays: dict[str, np.ndarray],
               bar_arrays: dict[str, dict[str, np.ndarray]],
               dom: DOMSnapshot | None,
           ) -> SignalOutput: ...

           async def initialize(self) -> None: ...
       ```

       Follow Phase 1 pattern: frozen dataclass with `__slots__`, `datetime.now(UTC)`.
       The Protocol uses structural typing per research recommendation (not ABC).
       `tick_arrays` matches TickBuffer.as_arrays() output keys: time_msc, bid, ask, last, spread, volume_real.
       `bar_arrays` is a dict of timeframe -> BarBuffer.as_arrays() output (e.g. bar_arrays["M1"]["close"]).
       `dom` is a DOMSnapshot or None for graceful degradation per D-13/FLOW-06.
