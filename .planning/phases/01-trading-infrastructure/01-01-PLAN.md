       class RiskConfig(BaseModel):
           aggressive_max: float = 100.0      # Equity ceiling for aggressive phase
           selective_max: float = 300.0        # Equity ceiling for selective phase
           aggressive_risk_pct: float = 0.10   # Per D-03: 10% risk per trade
           selective_risk_pct: float = 0.05    # Per D-03: 5% risk per trade
           conservative_risk_pct: float = 0.02 # Per D-03: 2% risk per trade
           daily_drawdown_pct: float = 0.05    # Per D-08: 3-5% daily limit
           weekly_drawdown_pct: float = 0.10   # Per RISK-07
           max_total_drawdown_pct: float = 0.25 # Per RISK-07
           max_consecutive_losses: int = 5     # Per D-08
           max_daily_trades: int = 20          # Per D-08
           rapid_equity_drop_pct: float = 0.05 # Per D-08
           rapid_equity_drop_window_minutes: int = 15 # Per D-08
           spread_threshold_multiplier: float = 2.0 # Per RISK-03
           spread_spike_multiplier: float = 5.0    # Per D-08
           spread_spike_duration_seconds: int = 30  # Per D-08

           def get_risk_pct(self, equity: float) -> float:
               """Return risk percentage for current capital phase per D-03."""
               if equity < self.aggressive_max:
                   return self.aggressive_risk_pct
               elif equity < self.selective_max:
                   return self.selective_risk_pct
               else:
                   return self.conservative_risk_pct

           Validators: aggressive_risk_pct/selective_risk_pct/conservative_risk_pct must be 0 < x <= 1.0.
           All pct fields use @field_validator to reject negative values and values > 1.0.

       class SessionConfig(BaseModel):
           windows: list[dict[str, str]] = [{"start": "13:00", "end": "17:00"}]  # Per RISK-06
           timezone: str = "UTC"
           reset_hour: int = 0  # Session boundary for counter reset per D-10

       class ExecutionConfig(BaseModel):
           symbol: str = "XAUUSD"
           magic_number: int = 20260327
           deviation: int = 20  # Slippage tolerance in points
           mode: str = "paper"  # Per D-01/D-02: "paper" or "live", manual switch only

           mt5_path: str | None = None
           mt5_login: int | None = None
           mt5_password: str | None = None
           mt5_server: str | None = None

           Validator: mode must be one of ("paper", "live")

       class DataConfig(BaseModel):
           tick_buffer_size: int = 10000

           storage_path: str = "data/"

       from dataclasses import dataclass, field
       from datetime import datetime
       from enum import Enum

       class EventType(str, Enum):
           TICK = "tick"
           BAR = "bar"
           DOM = "dom"
           FILL = "fill"
           ORDER_REJECTED = "order_rejected"
           POSITION_OPENED = "position_opened"
           POSITION_CLOSED = "position_closed"
           CIRCUIT_BREAKER_TRIPPED = "circuit_breaker_tripped"
           KILL_SWITCH_ACTIVATED = "kill_switch_activated"
           CONNECTION_LOST = "connection_lost"
           CONNECTION_RESTORED = "connection_restored"

       @dataclass(frozen=True, slots=True)
       class TickEvent:
           symbol: str
           time_msc: int          # Millisecond timestamp from MT5
           bid: float
           ask: float
           last: float
           volume: int
           flags: int
           volume_real: float
           spread: float          # Computed: ask - bid

       @dataclass(frozen=True, slots=True)
       class BarEvent:
           symbol: str
           timeframe: str         # "M1", "M5", "M15", "H1", "H4"
           time: int              # Unix timestamp
           open: float
           high: float
           low: float
           close: float
           tick_volume: int
           spread: int
           real_volume: int

       @dataclass(frozen=True, slots=True)
       class DOMEntry:
           type: int              # 1=sell, 2=buy (from MT5 BookInfo)
           price: float
           volume: int
           volume_dbl: float

       @dataclass(frozen=True, slots=True)
       class DOMSnapshot:
           symbol: str
           time_msc: int
           entries: tuple[DOMEntry, ...]  # Immutable tuple of DOM entries

       @dataclass(frozen=True, slots=True)
       class FillEvent:
           ticket: int
           symbol: str
           action: str            # "buy" or "sell"
           volume: float
           fill_price: float
           requested_price: float
           slippage: float
           sl: float
           tp: float | None
           magic: int
           is_paper: bool         # Per D-01: distinguishes paper vs live fills
           timestamp: datetime = field(default_factory=datetime.utcnow)
