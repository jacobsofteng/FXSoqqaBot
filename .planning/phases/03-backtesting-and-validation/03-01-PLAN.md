    ```python
    class LiveDataFeedAdapter:
        """Adapter wrapping MarketDataFeed + TickBuffer + BarBufferSet to implement DataFeedProtocol.

        Uses the adapter pattern (Research recommended) to keep existing live trading code untouched.
        The TradingEngine can use this adapter, and BacktestDataFeed implements the same Protocol directly.
        """
        def __init__(
            self,
            tick_buffer: TickBuffer,
            bar_buffers: BarBufferSet,
            symbol: str = "XAUUSD",
        ) -> None:
            self._tick_buffer = tick_buffer
            self._bar_buffers = bar_buffers
            self._symbol = symbol

        async def get_tick_arrays(self, symbol: str) -> dict[str, np.ndarray]:
            return self._tick_buffer.as_arrays()

        async def get_bar_arrays(self, symbol: str) -> dict[str, dict[str, np.ndarray]]:
            return {
                tf: self._bar_buffers[tf].as_arrays()
                for tf in self._bar_buffers.timeframes
            }

        async def get_dom(self, symbol: str) -> DOMSnapshot | None:
            return None  # DOM currently always None in live engine too

        def check_tick_freshness(self, max_age_seconds: float = 10.0) -> bool:
            latest = self._tick_buffer.latest
            if latest is None:
                return False
            age_ms = int(datetime.now(UTC).timestamp() * 1000) - latest.time_msc
            return age_ms < max_age_seconds * 1000
    ```
