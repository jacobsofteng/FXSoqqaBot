    ```python
    @dataclass(frozen=True, slots=True)
    class MonteCarloResult:
        """Result of Monte Carlo simulation per D-07."""
        n_simulations: int
        pct_5_equity: float       # 5th percentile final equity
        median_equity: float      # 50th percentile final equity
        pct_95_max_dd: float     # 95th percentile of max drawdowns (as fraction, e.g. 0.25 = 25%)
        p_value: float           # Fraction of runs that are net negative
        passes_threshold: bool   # Both D-07 criteria met
        equity_distribution: tuple[float, ...]  # All final equities for histogram (optional, can be truncated)

    def run_monte_carlo(
        trade_pnls: np.ndarray,
        starting_equity: float,
        n_simulations: int = 10_000,
        max_drawdown_threshold: float = 0.40,
        seed: int = 42,
    ) -> MonteCarloResult:
        """Monte Carlo trade sequence shuffling per D-07.

        Algorithm:
        1. Create numpy RNG with given seed for reproducibility
        2. For each of n_simulations iterations:
           a. Shuffle trade_pnls using rng.permutation()
           b. Build equity curve: starting_equity + cumsum(shuffled_pnls)
           c. Record final equity
           d. Compute max drawdown: max((running_max - equity) / running_max)
        3. Compute statistics:
           - pct_5_equity = np.percentile(final_equities, 5)
           - median_equity = np.percentile(final_equities, 50)
           - pct_95_max_dd = np.percentile(max_drawdowns, 95)
           - p_value = np.mean(final_equities < starting_equity)
        4. Evaluate D-07 dual threshold:
           - Criterion 1: pct_5_equity > starting_equity (5th percentile net positive)
           - Criterion 2: median_equity > starting_equity AND pct_95_max_dd < max_drawdown_threshold
           - passes = criterion_1 AND criterion_2

        Edge cases:
        - Zero trades: return MonteCarloResult(passes_threshold=False, p_value=1.0, ...)
        - Single trade: still shuffle (1 permutation = original), compute metrics

        Per Pitfall 5: acknowledge that shuffling breaks temporal dependencies.
        This is a known limitation documented in the result.
        """
    ```

    Vectorize the inner loop using numpy for performance:
    ```python
    # Vectorized approach: create (n_simulations, n_trades) matrix of shuffled indices
    # Then use fancy indexing to build all equity curves at once
    for i in range(n_simulations):
        shuffled = rng.permutation(trade_pnls)
        equity_curve = starting_equity + np.cumsum(shuffled)
        final_equities[i] = equity_curve[-1]
        running_max = np.maximum.accumulate(equity_curve)
        drawdowns = (running_max - equity_curve) / np.where(running_max > 0, running_max, 1.0)
        max_drawdowns[i] = drawdowns.max() if len(drawdowns) > 0 else 0.0
    ```

    ```python
    @dataclass(frozen=True, slots=True)
    class StressTestResult:
        """Result of Feigenbaum stress testing per TEST-06."""
        pre_transition_regime: str  # What regime was detected before transition
        transition_regime: str  # What regime was detected during transition
        post_transition_regime: str  # What regime was detected after transition
        pre_transition_detected: bool  # Was it classified as stable (TRENDING/RANGING)?
        transition_detected: bool  # Was PRE_BIFURCATION or HIGH_CHAOS detected?
        chaos_detected: bool  # Was HIGH_CHAOS detected post-transition?
        bifurcation_proximity_at_transition: float  # detect_bifurcation_proximity result
        passes: bool  # transition_detected AND chaos_detected

    class FeigenbaumStressTest:
        """Injects synthetic regime transitions to verify chaos module behavior per TEST-06.

        Generates synthetic price series with controlled period-doubling bifurcation:
        - Phase 1: Stable single-period oscillation
        - Phase 2: Period doubling (approaching Feigenbaum delta)
        - Phase 3: Chaotic regime (no clear period)

        Runs the chaos module over each phase and verifies it correctly classifies:
        - Pre-transition as stable (TRENDING or RANGING)
        - Transition as PRE_BIFURCATION or HIGH_CHAOS
        - Post-transition as HIGH_CHAOS
        """

        def __init__(self, config: ChaosConfig) -> None:
            self._config = config
            self._logger = structlog.get_logger().bind(component="stress_test")

        def generate_bifurcation_price_series(
            self,
            n_bars: int = 500,
            base_price: float = 2000.0,
            pre_transition_bars: int = 200,
            transition_bars: int = 100,
            post_transition_bars: int = 200,
            seed: int = 42,
        ) -> np.ndarray:
            """Generate synthetic price series with controlled bifurcation per Research code example.

            Phase 1 (pre_transition_bars): Regular oscillation
              prices[i] = base_price + 5.0 * sin(2*pi*i/20) + noise(0, 0.5)
            Phase 2 (transition_bars): Period doubling
              progress = (i - pre) / transition
              prices[i] = base + (1-progress)*5*sin(i/20) + progress*3*sin(i/10) + noise(0, 0.5 + progress*2)
            Phase 3 (post_transition_bars): Chaotic
              prices[i] = prices[i-1] + noise(0, 3.0) + 0.5*sin(random_phase)
            """

        async def run_stress_test(self) -> StressTestResult:
            """Run Feigenbaum stress test.

            1. Generate synthetic price series
            2. Build bar arrays from the series (OHLCV from price + small random wicks)
            3. Run chaos module over pre-transition, transition, and post-transition segments
            4. Check detect_bifurcation_proximity on the transition segment
            5. Report whether each phase was correctly classified
            """
    ```
